--タイマーを使う時は、いちばん最初のフレームでタイマー--キャストをパペットにしておきましょう。global dma,dmb,dmc,dmd,dme--dmaとdmbはユーザーの動作が正しいかどうかをチェック--dmcは何秒数えるか覚えておく変数--dmdはあと何秒か表示させるための変数（整数）on startMovie  initMovieendon exitFrame  if dmb <> 0 then    checkTimer  end if  endon checkTimer  set a = string(dmc - integer((the timer)/60))      if a <> dmd then    if the number of chars in a  = 3 then      set the castNum of sprite 15 = the number of cast ("Num" & (char 3 of a))      set the castNum of sprite 14 = the number of cast ("Num" & (char 2 of a))      set the castNum of sprite 13 = the number of cast ("Num" & (char 1 of a))      --のこり3桁秒の場合    else if the number of chars in a = 2 then      set the castNum of sprite 15 = the number of cast ("Num" & (char 2 of a))      set the castNum of sprite 14 = the number of cast ("Num" & (char 1 of a))      set the castNum of sprite 13 = the number of cast "NumNo"       --のこり2桁秒の場合    else      --のこり1桁秒の場合      set the castNum of sprite 15 = the number of cast ("Num" & a)      set the castNum of sprite 14 = the number of cast "NumNo"       set the castNum of sprite 13 = the number of cast "NumNo"       if a = "0" then        startTimer        set dmc = dmc + 10                if dmc > 180 then          go "hotelIn" of movie"hotel.dir"        else          set dma = 0          set dmb = 0          go "start"        end if              end if      set dmd = a    end if  end if  end
